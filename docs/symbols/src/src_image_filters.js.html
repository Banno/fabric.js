 10 fabric.Image.filters.Grayscale = fabric.util.createClass( /** @scope fabric.Image.filters.Grayscale.prototype */ {
 11   
 12   /**
 13    * @param {String} type
 14    */
 15   type: "Grayscale",
 16   
 17   /** 
 18    * @method applyTo
 19    * @memberOf fabric.Image.filters.Grayscale.prototype
 20    * @param {Object} canvasEl Canvas element to apply filter to
 21    */
 22   applyTo: function(canvasEl) {
 23     var context = canvasEl.getContext('2d'),
 24         imageData = context.getImageData(0, 0, canvasEl.width, canvasEl.height),
 25         data = imageData.data, 
 26         iLen = imageData.width,
 27         jLen = imageData.height,
 28         index, average, i, j;
 29 
 30      for (i = 0; i < iLen; i++) {
 31        for (j = 0; j < jLen; j++) {
 32 
 33          index = (i * 4) * jLen + (j * 4);
 34          average = (data[index] + data[index + 1] + data[index + 2]) / 3;
 35 
 36          data[index]     = average;
 37          data[index + 1] = average;
 38          data[index + 2] = average;
 39        }
 40      }
 41 
 42      context.putImageData(imageData, 0, 0);
 43   },
 44   
 45   /** 
 46    * @method toJSON
 47    * @return {String} json representation of filter
 48    */
 49   toJSON: function() {
 50     return { type: this.type };
 51   }
 52 });
 53 
 54 fabric.Image.filters.Grayscale.fromObject = function() {
 55   return new fabric.Image.filters.Grayscale();
 56 };

 59  * @class fabric.Image.filters.RemoveWhite
 60  * @memberOf fabric.Image.filters
 61  */
 62 fabric.Image.filters.RemoveWhite = fabric.util.createClass( /** @scope fabric.Image.filters.RemoveWhite.prototype */ {
 63   
 64   /**
 65    * @param {String} type
 66    */
 67   type: "RemoveWhite",
 68   
 69   /** 
 70    * @memberOf fabric.Image.filters.RemoveWhite.prototype
 71    * @param {Object} [options] Options object
 72    */
 73   initialize: function(options) {
 74     options || (options = { });
 75     this.threshold = options.threshold || 30;
 76     this.distance = options.distance || 20;
 77   },
 78   
 79   /** 
 80    * @method applyTo
 81    * @param {Object} canvasEl Canvas element to apply filter to
 82    */
 83   applyTo: function(canvasEl) {
 84     var context = canvasEl.getContext('2d'),
 85         imageData = context.getImageData(0, 0, canvasEl.width, canvasEl.height),
 86         data = imageData.data,
 87         threshold = this.threshold,
 88         distance = this.distance,
 89         limit = 255 - threshold,
 90         abs = Math.abs,
 91         r, g, b;
 92 
 93     for (var i = 0, len = data.length; i < len; i += 4) {
 94 
 95       r = data[i];
 96       g = data[i+1];
 97       b = data[i+2];
 98 
 99       if (r > limit && 
100           g > limit && 
101           b > limit && 
102           abs(r-g) < distance && 
103           abs(r-b) < distance &&
104           abs(g-b) < distance) {
105 
106         data[i+3] = 1;
107       }
108     }
109 
110     context.putImageData(imageData, 0, 0);
111   },
112   
113   /** 
114    * @method toJSON
115    * @return {String} json representation of filter
116    */
117   toJSON: function() {
118     return { 
119       type: this.type, 
120       threshold: this.threshold, 
121       distance: this.distance
122     };
123   }
124 });
125 
126 fabric.Image.filters.RemoveWhite.fromObject = function(object) {
127   return new fabric.Image.filters.RemoveWhite(object);
